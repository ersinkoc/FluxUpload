Quick Start

Get up and running in minutes

Installation 

npm install fluxupload

Basic Usage 

const FluxUpload = require('fluxupload');

const uploader = new FluxUpload({
  storage: 'local',
  destination: './uploads'
});

server.on('request', (req, res) => {
  uploader.handle(req)
    .then(result => {
      res.end(JSON.stringify(result));
    });
});

Full Example

Complete setup with security and validation

const http = require('http');
const FluxUpload = require('fluxupload');
const {
  LocalStorage,
  CsrfProtection,
  RateLimiter,
  QuotaLimiter,
  MagicByteDetector,
  StreamHasher
} = FluxUpload;

// Create uploader with plugins
const uploader = new FluxUpload({
  storage: new LocalStorage({
    destination: './uploads',
    naming: 'uuid'
  }),

  validators: [
    new CsrfProtection(),
    new RateLimiter({ limit: 100, windowMs: 60000 }),
    new QuotaLimiter({ maxFileSize: 100 * 1024 * 1024 }),
    new MagicByteDetector()
  ],

  transformers: [
    new StreamHasher({ algorithm: 'sha256' })
  ],

  limits: {
    fileSize: 100 * 1024 * 1024,  // 100MB
    files: 10
  }
});

// Initialize
await uploader.initialize();

// Handle uploads
http.createServer(async (req, res) => {
  if (req.url === '/upload' && req.method === 'POST') {
    try {
      const result = await uploader.handle(req);
      res.writeHead(200, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({
        success: true,
        files: result.files,
        fields: result.fields
      }));
    } catch (error) {
      res.writeHead(400, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({ error: error.message }));
    }
  }
}).listen(3000);

console.log('Server running on http://localhost:3000');
